# ==================== 安装建议 ====================
# 1. 基础安装（最小依赖）:
#    pip install numpy pandas scikit-learn matplotlib seaborn joblib
#
# 2. 完整安装（包括高级算法）:
#    pip install -r requirements.txt
#
# 3. 不安装Jupyter（如果只需要运行脚本）:
#    pip install numpy pandas scikit-learn xgboost lightgbm imbalanced-learn matplotlib seaborn joblib tqdm
#
# 4. 使用conda安装（推荐，避免兼容性问题）:
#    conda create -n churn_prediction python=3.10
#    conda activate churn_prediction
#    pip install -r requirements.txt
#
# 5. 在虚拟环境中安装（推荐）:
#    python -m venv venv
#    source venv/bin/activate  # Linux/Mac
#    # 或 venv\Scripts\activate  # Windows
#    pip install -r requirements.txt

# Python版本: 3.8 或更高
# 推荐使用: Python 3.10 或 3.11
